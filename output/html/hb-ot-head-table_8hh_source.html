50 unsigned int get_upem () const
51 {
52 unsigned int upem = unitsPerEm;
53 /* If no valid head table found, assume 1000, which matches typical Type1 usage. */
54 return 16 <= upem && upem <= 16384 ? upem : 1000;
55 }

92 protected:
93 FixedVersion<>version; /* Version of the head table--currently
94 * 0x00010000u for version 1.0. */
95 FixedVersion<>fontRevision; /* Set by font manufacturer. */
96 HBUINT32 checkSumAdjustment; /* To compute: set it to 0, sum the
97 * entire font as HBUINT32, then store
98 * 0xB1B0AFBAu - sum. */
99 HBUINT32 magicNumber; /* Set to 0x5F0F3CF5u. */
100 HBUINT16 flags; /* Bit 0: Baseline for font at y=0;
101 * Bit 1: Left sidebearing point at x=0;
102 * Bit 2: Instructions may depend on point size;
103 * Bit 3: Force ppem to integer values for all
104 * internal scaler math; may use fractional
105 * ppem sizes if this bit is clear;
106 * Bit 4: Instructions may alter advance width
107 * (the advance widths might not scale linearly);
108 * Bits 5-10: These should be set according to
109 * Apple's specification. However, they are not
110 * implemented in OpenType.
111 * Bit 5: This bit should be set in fonts that are
112 * intended to e laid out vertically, and in
113 * which the glyphs have been drawn such that an
114 * x-coordinate of 0 corresponds to the desired
115 * vertical baseline.
116 * Bit 6: This bit must be set to zero.
117 * Bit 7: This bit should be set if the font
118 * requires layout for correct linguistic
119 * rendering (e.g. Arabic fonts).
120 * Bit 8: This bit should be set for a GX font
121 * which has one or more metamorphosis effects
122 * designated as happening by default.
123 * Bit 9: This bit should be set if the font
124 * contains any strong right-to-left glyphs.
125 * Bit 10: This bit should be set if the font
126 * contains Indic-style rearrangement effects.
127 * Bit 11: Font data is 'lossless,' as a result
128 * of having been compressed and decompressed
129 * with the Agfa MicroType Express engine.
130 * Bit 12: Font converted (produce compatible metrics)
131 * Bit 13: Font optimized for ClearType™.
132 * Note, fonts that rely on embedded bitmaps (EBDT)
133 * for rendering should not be considered optimized
134 * for ClearType, and therefore should keep this bit
135 * cleared.
136 * Bit 14: Last Resort font. If set, indicates that
137 * the glyphs encoded in the cmap subtables are simply
138 * generic symbolic representations of code point
139 * ranges and don’t truly represent support for those
140 * code points. If unset, indicates that the glyphs
141 * encoded in the cmap subtables represent proper
142 * support for those code points.
143 * Bit 15: Reserved, set to 0. */
144 HBUINT16 unitsPerEm; /* Valid range is from 16 to 16384. This value
145 * should be a power of 2 for fonts that have
146 * TrueType outlines. */
147 LONGDATETIME created; /* Number of seconds since 12:00 midnight,
148 January 1, 1904. 64-bit integer */
149 LONGDATETIME modified; /* Number of seconds since 12:00 midnight,
150 January 1, 1904. 64-bit integer */
151 HBINT16 xMin; /* For all glyph bounding boxes. */
152 HBINT16 yMin; /* For all glyph bounding boxes. */
153 HBINT16 xMax; /* For all glyph bounding boxes. */
154 HBINT16 yMax; /* For all glyph bounding boxes. */
155 HBUINT16 macStyle; /* Bit 0: Bold (if set to 1);
156 * Bit 1: Italic (if set to 1)
157 * Bit 2: Underline (if set to 1)
158 * Bit 3: Outline (if set to 1)
159 * Bit 4: Shadow (if set to 1)
160 * Bit 5: Condensed (if set to 1)
161 * Bit 6: Extended (if set to 1)
162 * Bits 7-15: Reserved (set to 0). */
163 HBUINT16 lowestRecPPEM; /* Smallest readable size in pixels. */
164 HBINT16 fontDirectionHint; /* Deprecated (Set to 2).
165 * 0: Fully mixed directional glyphs;
166 * 1: Only strongly left to right;
167 * 2: Like 1 but also contains neutrals;
168 * -1: Only strongly right to left;
169 * -2: Like -1 but also contains neutrals. */
170 public:
171 HBUINT16 indexToLocFormat; /* 0 for short offsets, 1 for long. */
172 HBUINT16 glyphDataFormat; /* 0 for current format. */
173
174 DEFINE_SIZE_STATIC (54);
