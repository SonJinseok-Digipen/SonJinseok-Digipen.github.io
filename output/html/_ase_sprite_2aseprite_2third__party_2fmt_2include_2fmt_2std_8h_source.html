113template <typename T>
114using variant_index_sequence =
115 std::make_index_sequence<std::variant_size<T>::value>;
116
117template <typename> struct is_variant_like_ : std::false_type {};
118template <typename... Types>
119struct is_variant_like_<std::variant<Types...>> : std::true_type {};

121// formattable element check.
122template <typename T, typename C> class is_variant_formattable_ {
123 template <std::size_t... Is>
124 static std::conjunction<
125 is_formattable<std::variant_alternative_t<Is, T>, C>...>
126 check(std::index_sequence<Is...>);
127
128 public:
129 static constexpr const bool value =
130 decltype(check(variant_index_sequence<T>{}))::value;
131};

133template <typename Char, typename OutputIt, typename T>
134auto write_variant_alternative(OutputIt out, const T& v) -> OutputIt {
135 if constexpr (is_string<T>::value)
136 return write_escaped_string<Char>(out, detail::to_string_view(v));
137 else if constexpr (std::is_same_v<T, Char>)
138 return write_escaped_char(out, v);
139 else
140 return write<Char>(out, v);
141}

143} // namespace detail

145template <typename T> struct is_variant_like {

154template <typename Variant, typename Char>
155struct formatter<
156 Variant, Char,
157 std::enable_if_t<std::conjunction_v<
158 is_variant_like<Variant>, is_variant_formattable<Variant, Char>>>> {
159 template <typename ParseContext>
160 FMT_CONSTEXPR auto parse(ParseContext& ctx) -> decltype(ctx.begin()) {
161 return ctx.begin();
162 }
163
164 template <typename FormatContext>
165 auto format(const Variant& value, FormatContext& ctx) const
166 -> decltype(ctx.out()) {
167 auto out = ctx.out();
168
169 out = detail::write<Char>(out, "variant(");
170 std::visit(
171 [&](const auto& v) {
172 out = detail::write_variant_alternative<Char>(out, v);
173 },
174 value);
175 *out++ = ')';
176 return out;
177 }
178};

183template <typename T, typename Char>
184struct formatter<
185 T, Char,
186 typename std::enable_if<std::is_base_of<std::exception, T>::value>::type> {
187 private:
188 bool with_typename_ = false;
189
190 public:
191 FMT_CONSTEXPR auto parse(basic_format_parse_context<Char>& ctx)
192 -> decltype(ctx.begin()) {
193 auto it = ctx.begin();
194 auto end = ctx.end();
195 if (it == end || *it == '}') return it;
196 if (*it == 't') {
197 ++it;
198 with_typename_ = true;
199 }
200 return it;
201 }
202
203 template <typename OutputIt>
204 auto format(const std::exception& ex,
205 basic_format_context<OutputIt, Char>& ctx) const -> OutputIt {
206 basic_format_specs<Char> spec;
207 auto out = ctx.out();
208 if (!with_typename_)
209 return detail::write_bytes(out, string_view(ex.what()), spec);
210
211 const std::type_info& ti = typeid(ex);
212#ifdef FMT_HAS_ABI_CXA_DEMANGLE
213 int status = 0;
214 std::size_t size = 0;
215 std::unique_ptr<char, decltype(&std::free)> demangled_name_ptr(
216 abi::__cxa_demangle(ti.name(), nullptr, &size, &status), &std::free);
217 out = detail::write_bytes(
218 out,
219 string_view(demangled_name_ptr ? demangled_name_ptr.get() : ti.name()),
220 spec);
221#elif FMT_MSC_VERSION
222 string_view demangled_name_view(ti.name());
223 if (demangled_name_view.starts_with("class "))
224 demangled_name_view.remove_prefix(6);
225 else if (demangled_name_view.starts_with("struct "))
226 demangled_name_view.remove_prefix(7);
227 out = detail::write_bytes(out, demangled_name_view, spec);
228#else
229 out = detail::write_bytes(out, string_view(ti.name()), spec);
230#endif
231 out = detail::write<Char>(out, Char(':'));
232 out = detail::write<Char>(out, Char(' '));
233 out = detail::write_bytes(out, string_view(ex.what()), spec);
234
235 return out;
236 }
237};
