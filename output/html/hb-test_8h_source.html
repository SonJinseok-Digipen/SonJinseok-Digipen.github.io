142/* Drops the "test_" prefix and converts '_' to '/'.
143 * Essentially builds test path from function name. */
144static inline char *
145hb_test_normalize_path (const char *path)
146{
147 char *s, *p;
148
149 g_assert (0 == strncmp (path, "test_", 5));
150 path += 4;
151
152 s = g_strdup (path);
153 for (p = s; *p; p++)
154 if (*p == '_')
155 *p = '/';
156
157 return s;
158}

181static inline void hb_test_assert_blobs_equal (hb_blob_t *expected_blob, hb_blob_t *actual_blob)
182{
183 unsigned int expected_length, actual_length;
184 const char *raw_expected = hb_blob_get_data (expected_blob, &expected_length);
185 const char *raw_actual = hb_blob_get_data (actual_blob, &actual_length);
186 g_assert_cmpint(expected_length, ==, actual_length);
187 if (memcmp (raw_expected, raw_actual, expected_length) != 0)
188 {
189 for (unsigned int i = 0; i < expected_length; i++)
190 {
191 int expected = *(raw_expected + i);
192 int actual = *(raw_actual + i);
193 if (expected != actual) fprintf(stderr, "+%u %02x != %02x\n", i, expected, actual);
194 else fprintf(stderr, "+%u %02x\n", i, expected);
195 }
196 }
197 g_assert_cmpint(0, ==, memcmp(raw_expected, raw_actual, expected_length));
198}

301static inline hb_face_t *
302hb_test_open_font_file (const char *font_path)
303{
304#if GLIB_CHECK_VERSION(2,37,2)
305 char *path = g_test_build_filename (G_TEST_DIST, font_path, NULL);
306#else
307 char *path = g_strdup (font_path);
308#endif
309
310 hb_blob_t *blob = hb_blob_create_from_file_or_fail (path);
311 hb_face_t *face;
312 if (!blob)
313 g_error ("Font %s not found.", path);
314
315 face = hb_face_create (blob, 0);
316 hb_blob_destroy (blob);
317
318 g_free (path);
319
320 return face;
321}
