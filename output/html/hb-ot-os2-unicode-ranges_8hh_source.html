222static unsigned int
223_hb_ot_os2_get_unicode_range_bit (hb_codepoint_t cp)
224{
225 auto *range = hb_sorted_array (_hb_os2_unicode_ranges).bsearch (cp);
226 return range ? range->bit : -1;
227}
