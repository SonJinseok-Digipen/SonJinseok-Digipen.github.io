71 int auxA = (((int)m_a) % m_pixelCount > m_pixelCount / 2) ? 1: 0;
72 return rgba(int(m_r / m_pixelCount + auxR),
73 int(m_g / m_pixelCount + auxG),
74 int(m_b / m_pixelCount + auxB),
75 int(m_a / m_pixelCount + auxA));
