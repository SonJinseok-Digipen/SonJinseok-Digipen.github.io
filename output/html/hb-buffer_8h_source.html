709typedef enum { /*< flags >*/
710 HB_BUFFER_DIFF_FLAG_EQUAL = 0x0000,
711
712 /* Buffers with different content_type cannot be meaningfully compared
713 * in any further detail. */
714 HB_BUFFER_DIFF_FLAG_CONTENT_TYPE_MISMATCH = 0x0001,
715
716 /* For buffers with differing length, the per-glyph comparison is not
717 * attempted, though we do still scan reference for dottedcircle / .notdef
718 * glyphs. */
719 HB_BUFFER_DIFF_FLAG_LENGTH_MISMATCH = 0x0002,
720
721 /* We want to know if dottedcircle / .notdef glyphs are present in the
722 * reference, as we may not care so much about other differences in this
723 * case. */
724 HB_BUFFER_DIFF_FLAG_NOTDEF_PRESENT = 0x0004,
725 HB_BUFFER_DIFF_FLAG_DOTTED_CIRCLE_PRESENT = 0x0008,
726
727 /* If the buffers have the same length, we compare them glyph-by-glyph
728 * and report which aspect(s) of the glyph info/position are different. */
729 HB_BUFFER_DIFF_FLAG_CODEPOINT_MISMATCH = 0x0010,
730 HB_BUFFER_DIFF_FLAG_CLUSTER_MISMATCH = 0x0020,
731 HB_BUFFER_DIFF_FLAG_GLYPH_FLAGS_MISMATCH = 0x0040,
732 HB_BUFFER_DIFF_FLAG_POSITION_MISMATCH = 0x0080
733
734} hb_buffer_diff_flags_t;
