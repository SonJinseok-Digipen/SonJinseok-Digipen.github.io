141template <typename Data, unsigned int WheresData>
142struct hb_data_wrapper_t
143{
144 static_assert (WheresData > 0, "");
145
146 Data * get_data () const
147 { return *(((Data **) (void *) this) - WheresData); }
148
149 bool is_inert () const { return !get_data (); }
150
151 template <typename Stored, typename Subclass>
152 Stored * call_create () const { return Subclass::create (get_data ()); }
153};
154template <>
155struct hb_data_wrapper_t<void, 0>
156{
157 bool is_inert () const { return false; }
158
159 template <typename Stored, typename Funcs>
160 Stored * call_create () const { return Funcs::create (); }
161};

166template <typename Returned,
167 typename Subclass = void,
168 typename Data = void,
169 unsigned int WheresData = 0,
170 typename Stored = Returned>
171struct hb_lazy_loader_t : hb_data_wrapper_t<Data, WheresData>
172{
173 typedef typename hb_non_void_t<Subclass,
174 hb_lazy_loader_t<Returned,Subclass,Data,WheresData,Stored>
175 >::value Funcs;
176
177 void init0 () {} /* Init, when memory is already set to 0. No-op for us. */
178 void init () { instance.set_relaxed (nullptr); }
179 void fini () { do_destroy (instance.get ()); }
180
181 void free_instance ()
182 {
183 retry:
184 Stored *p = instance.get ();
185 if (unlikely (p && !cmpexch (p, nullptr)))
186 goto retry;
187 do_destroy (p);
188 }
189
190 static void do_destroy (Stored *p)
191 {
192 if (p && p != const_cast<Stored *> (Funcs::get_null ()))
193 Funcs::destroy (p);
194 }
195
196 const Returned * operator -> () const { return get (); }
197 template <typename U = Returned, hb_enable_if (!hb_is_same (U, void))>
198 const U & operator * () const { return *get (); }
199 explicit operator bool () const
200 { return get_stored () != Funcs::get_null (); }
201 template <typename C> operator const C * () const { return get (); }
202
203 Stored * get_stored () const
204 {
205 retry:
206 Stored *p = this->instance.get ();
207 if (unlikely (!p))
208 {
209 if (unlikely (this->is_inert ()))
210 return const_cast<Stored *> (Funcs::get_null ());
211
212 p = this->template call_create<Stored, Funcs> ();
213 if (unlikely (!p))
214 p = const_cast<Stored *> (Funcs::get_null ());
215
216 if (unlikely (!cmpexch (nullptr, p)))
217 {
218 do_destroy (p);
219 goto retry;
220 }
221 }
222 return p;
223 }
224 Stored * get_stored_relaxed () const
225 {
226 return this->instance.get_relaxed ();
227 }
228
229 bool cmpexch (Stored *current, Stored *value) const
230 {
231 /* This *must* be called when there are no other threads accessing. */
232 return this->instance.cmpexch (current, value);
233 }
234
235 const Returned * get () const { return Funcs::convert (get_stored ()); }
236 const Returned * get_relaxed () const { return Funcs::convert (get_stored_relaxed ()); }
237 Returned * get_unconst () const { return const_cast<Returned *> (Funcs::convert (get_stored ())); }
238
239 /* To be possibly overloaded by subclasses. */
240 static Returned* convert (Stored *p) { return p; }
241
242 /* By default null/init/fini the object. */
243 static const Stored* get_null () { return &Null (Stored); }
244 static Stored *create (Data *data)
245 {
246 Stored *p = (Stored *) hb_calloc (1, sizeof (Stored));
247 if (likely (p))
248 p = new (p) Stored (data);
249 return p;
250 }
251 static Stored *create ()
252 {
253 Stored *p = (Stored *) hb_calloc (1, sizeof (Stored));
254 if (likely (p))
255 p = new (p) Stored ();
256 return p;
257 }
258 static void destroy (Stored *p)
259 {
260 p->~Stored ();
261 hb_free (p);
262 }
263
264// private:
265 /* Must only have one pointer. */
266 hb_atomic_ptr_t<Stored *> instance;
267};
