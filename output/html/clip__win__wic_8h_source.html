18// Successful calls to CoInitialize() (S_OK or S_FALSE) must match
19// the calls to CoUninitialize().
20// From: https://docs.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-couninitialize#remarks
21struct coinit {
22 HRESULT hr;
23 coinit() {
24 hr = CoInitialize(nullptr);
25 }
26 ~coinit() {
27 if (hr == S_OK || hr == S_FALSE)
28 CoUninitialize();
29 }
30};

32template<class T>
33class comptr {
34public:
35 comptr() { }
36 explicit comptr(T* ptr) : m_ptr(ptr) { }
37 comptr(const comptr&) = delete;
38 comptr& operator=(const comptr&) = delete;
39 ~comptr() { reset(); }
40
41 T** operator&() { return &m_ptr; }
42 T* operator->() { return m_ptr; }
43 bool operator!() const { return !m_ptr; }
44
45 T* get() { return m_ptr; }
46 void reset() {
47 if (m_ptr) {
48 m_ptr->Release();
49 m_ptr = nullptr;
50 }
51 }
52private:
53 T* m_ptr = nullptr;
54};

59bool write_png_on_stream(const image& image,
60 IStream* stream) {
61 const image_spec& spec = image.spec();
62
63 comptr<IWICBitmapEncoder> encoder;
64 HRESULT hr = CoCreateInstance(CLSID_WICPngEncoder,
65 nullptr, CLSCTX_INPROC_SERVER,
66 IID_PPV_ARGS(&encoder));
67 if (FAILED(hr))
68 return false;
69
70 hr = encoder->Initialize(stream, WICBitmapEncoderNoCache);
71 if (FAILED(hr))
72 return false;
73
74 comptr<IWICBitmapFrameEncode> frame;
75 comptr<IPropertyBag2> options;
76 hr = encoder->CreateNewFrame(&frame, &options);
77 if (FAILED(hr))
78 return false;
79
80 hr = frame->Initialize(options.get());
81 if (FAILED(hr))
82 return false;
83
84 // PNG encoder (and decoder) only supports GUID_WICPixelFormat32bppBGRA for 32bpp.
85 // See: https://docs.microsoft.com/en-us/windows/win32/wic/-wic-codec-native-pixel-formats#png-native-codec
86 WICPixelFormatGUID pixelFormat = GUID_WICPixelFormat32bppBGRA;
87 hr = frame->SetPixelFormat(&pixelFormat);
88 if (FAILED(hr))
89 return false;
90
91 hr = frame->SetSize(spec.width, spec.height);
92 if (FAILED(hr))
93 return false;
94
95 std::vector<uint32_t> buf;
96 uint8_t* ptr = (uint8_t*)image.data();
97 int bytes_per_row = spec.bytes_per_row;
98
99 // Convert to GUID_WICPixelFormat32bppBGRA if needed
100 if (spec.red_mask != 0xff0000 ||
101 spec.green_mask != 0xff00 ||
102 spec.blue_mask != 0xff ||
103 spec.alpha_mask != 0xff000000) {
104 buf.resize(spec.width * spec.height);
105 uint32_t* dst = (uint32_t*)&buf[0];
106 uint32_t* src = (uint32_t*)image.data();
107 for (int y=0; y<spec.height; ++y) {
108 auto src_line_start = src;
109 for (int x=0; x<spec.width; ++x) {
110 uint32_t c = *src;
111 *dst = ((((c & spec.red_mask ) >> spec.red_shift ) << 16) |
112 (((c & spec.green_mask) >> spec.green_shift) << 8) |
113 (((c & spec.blue_mask ) >> spec.blue_shift ) ) |
114 (((c & spec.alpha_mask) >> spec.alpha_shift) << 24));
115 ++dst;
116 ++src;
117 }
118 src = (uint32_t*)(((uint8_t*)src_line_start) + spec.bytes_per_row);
119 }
120 ptr = (uint8_t*)&buf[0];
121 bytes_per_row = 4 * spec.width;
122 }
123
124 hr = frame->WritePixels(spec.height,
125 bytes_per_row,
126 bytes_per_row * spec.height,
127 (BYTE*)ptr);
128 if (FAILED(hr))
129 return false;
130
131 hr = frame->Commit();
132 if (FAILED(hr))
133 return false;
134
135 hr = encoder->Commit();
136 if (FAILED(hr))
137 return false;
138
139 return true;
140}

142HGLOBAL write_png(const image& image) {
143 coinit com;
144
145 comptr<IStream> stream;
146 HRESULT hr = CreateStreamOnHGlobal(nullptr, false, &stream);
147 if (FAILED(hr))
148 return nullptr;
149
150 bool result = write_png_on_stream(image, stream.get());
151
152 HGLOBAL handle;
153 hr = GetHGlobalFromStream(stream.get(), &handle);
154 if (result)
155 return handle;
156
157 GlobalFree(handle);
158 return nullptr;
159}

162// Decode the clipboard data from PNG format
163
164bool read_png(const uint8_t* buf,
165 const UINT len,
166 image* output_image,
167 image_spec* output_spec) {
168 coinit com;
169
170 comptr<IStream> stream(SHCreateMemStream(buf, len));
171 if (!stream)
172 return false;
173
174 comptr<IWICBitmapDecoder> decoder;
175 HRESULT hr = CoCreateInstance(CLSID_WICPngDecoder2,
176 nullptr, CLSCTX_INPROC_SERVER,
177 IID_PPV_ARGS(&decoder));
178 if (FAILED(hr)) {
179 hr = CoCreateInstance(CLSID_WICPngDecoder1,
180 nullptr, CLSCTX_INPROC_SERVER,
181 IID_PPV_ARGS(&decoder));
182 if (FAILED(hr))
183 return false;
184 }
185
186 // Can decoder be nullptr if hr is S_OK/successful? We've received
187 // some crash reports that might indicate this.
188 if (!decoder)
189 return false;
190
191 hr = decoder->Initialize(stream.get(), WICDecodeMetadataCacheOnDemand);
192 if (FAILED(hr))
193 return false;
194
195 comptr<IWICBitmapFrameDecode> frame;
196 hr = decoder->GetFrame(0, &frame);
197 if (FAILED(hr))
198 return false;
199
200 WICPixelFormatGUID pixelFormat;
201 hr = frame->GetPixelFormat(&pixelFormat);
202 if (FAILED(hr))
203 return false;
204
205 // Only support this pixel format
206 // TODO add support for more pixel formats
207 if (pixelFormat != GUID_WICPixelFormat32bppBGRA)
208 return false;
209
210 UINT width = 0, height = 0;
211 hr = frame->GetSize(&width, &height);
212 if (FAILED(hr))
213 return false;
214
215 image_spec spec;
216 spec.width = width;
217 spec.height = height;
218 spec.bits_per_pixel = 32;
219 spec.bytes_per_row = 4 * width;
220 spec.red_mask = 0xff0000;
221 spec.green_mask = 0xff00;
222 spec.blue_mask = 0xff;
223 spec.alpha_mask = 0xff000000;
224 spec.red_shift = 16;
225 spec.green_shift = 8;
226 spec.blue_shift = 0;
227 spec.alpha_shift = 24;
228
229 if (output_spec)
230 *output_spec = spec;
231
232 if (output_image) {
233 image img(spec);
234
235 hr = frame->CopyPixels(
236 nullptr, // Entire bitmap
237 spec.bytes_per_row,
238 spec.bytes_per_row * spec.height,
239 (BYTE*)img.data());
240 if (FAILED(hr)) {
241 return false;
242 }
243
244 std::swap(*output_image, img);
245 }
246
247 return true;
248}
