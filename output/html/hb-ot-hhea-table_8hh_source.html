56 public:
57 FixedVersion<>version; /* 0x00010000u for version 1.0. */
58 FWORD ascender; /* Typographic ascent. */
59 FWORD descender; /* Typographic descent. */
60 FWORD lineGap; /* Typographic line gap. */
61 UFWORD advanceMax; /* Maximum advance width/height value in
62 * metrics table. */
63 FWORD minLeadingBearing;
64 /* Minimum left/top sidebearing value in
65 * metrics table. */
66 FWORD minTrailingBearing;
67 /* Minimum right/bottom sidebearing value;
68 * calculated as Min(aw - lsb -
69 * (xMax - xMin)) for horizontal. */
70 FWORD maxExtent; /* horizontal: Max(lsb + (xMax - xMin)),
71 * vertical: minLeadingBearing+(yMax-yMin). */
72 HBINT16 caretSlopeRise; /* Used to calculate the slope of the
73 * cursor (rise/run); 1 for vertical caret,
74 * 0 for horizontal.*/
75 HBINT16 caretSlopeRun; /* 0 for vertical caret, 1 for horizontal. */
76 HBINT16 caretOffset; /* The amount by which a slanted
77 * highlight on a glyph needs
78 * to be shifted to produce the
79 * best appearance. Set to 0 for
80 * non-slanted fonts. */
81 HBINT16 reserved1; /* Set to 0. */
82 HBINT16 reserved2; /* Set to 0. */
83 HBINT16 reserved3; /* Set to 0. */
84 HBINT16 reserved4; /* Set to 0. */
85 HBINT16 metricDataFormat;/* 0 for current format. */
86 HBUINT16 numberOfLongMetrics;
87 /* Number of LongMetric entries in metric
88 * table. */
89 public:
