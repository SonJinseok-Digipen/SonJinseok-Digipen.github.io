52// Sets dlg tags on its construction and removes them on its destruction.
53// Instead of explicitly constructing an object, just use the dlg_tags and
54// dlg_tags_global macros which will construct one in the current scope.
55// Just forwards the arguments on construction to dlg_add_tag, so if func
56// is nullptr the tags will be applied even to called functions from the current
57// scope, otherwise only to calls coming directly from the current function.
58class TagsGuard {
59public:
60 TagsGuard(const char** tags, const char* func) : tags_(tags), func_(func) {
61 while(*tags) {
62 dlg_add_tag(*tags, func);
63 ++tags;
64 }
65 }
66
67 ~TagsGuard() {
68 while(*tags_) {
69 dlg_remove_tag(*tags_, func_);
70 ++tags_;
71 }
72 }
73
74protected:
75 const char** tags_;
76 const char* func_;
77};

190// Implements std::basic_streambuf into the dlg_thread_buffer.
191class StreamBuffer : public std::basic_streambuf<char> {
192public:
193 StreamBuffer(char*& buf, std::size_t& size) : buf_(buf), size_(size) {
194 setp(buf, buf + size); // we will only read from it
195 }
196
197 ~StreamBuffer() {
198 // make sure only the end has a null terminator
199 setp(std::remove(buf_, pptr(), '\0'), buf_ + size_);
200 sputc('\0');
201 }
202
203 int_type overflow(int_type ch = traits_type::eof()) override {
204 if(pptr() >= epptr()) {
205 auto off = pptr() - buf_;
206 size_ = size_ * 2 + 1;
207 buf_ = static_cast<char*>(std::realloc(buf_, size_));
208 if(!buf_) {
209 size_ = 0;
210 setp(nullptr, nullptr);
211 return traits_type::eof();
212 }
213
214 setp(buf_ + off, buf_ + size_);
215 }
216
217 if(!traits_type::eq_int_type(ch, traits_type::eof())) {
218 *pptr() = (char_type) ch;
219 pbump(1);
220 }
221
222 return 0;
223 }
224
225protected:
226 char*& buf_;
227 size_t& size_;
228};

230// Like std::strstr but only matches if target is not wrapped in backslashes,
231// otherwise prints the target.
232inline const char* find_next(std::ostream& os, const char*& src, const char* target) {
233 auto len = std::strlen(target);
234 const char* next = std::strstr(src, target);
235 while(next && next > src && *(next - 1) == '\\' && *(next + len) == '\\') {
236 os.write(src, next - src - 1);
237 os.write(target, len);
238 src = next + len + 1;
239 next = std::strstr(next + 1, target);
240 }
241
242 return next;
243}

247template<typename... Args>
248const char* tlformat(StringParam fmt, Args&&... args) {
249 {
250 std::size_t* size;
251 char** dbuf = dlg_thread_buffer(&size);
252 detail::StreamBuffer buf(*dbuf, *size);
253 std::ostream output(&buf);
254 gformat(output, DLG_FORMAT_DEFAULT_REPLACE, fmt, std::forward<Args>(args)...);

276void gformat(std::ostream& os, StringParam replace, StringParam fmt) {
277 if(detail::find_next(os, fmt.str, replace.str)) {
278 throw std::invalid_argument("Too few arguments given to format");
279 }
280
281 os << fmt.str;
282}

284template<typename Arg, typename... Args>
285void gformat(std::ostream& os, StringParam replace, StringParam fmt, Arg&& arg, Args&&... args) {
286 const char* next = detail::find_next(os, fmt.str, replace.str);
287 if(!next) {
288 throw std::invalid_argument("Too many arguments to format supplied");
289 }
290
291 // XXX: any drawback compared to formatted?
292 // this allows to use temporary ostream modifiers only to real arguments
293 os.write(fmt.str, next - fmt.str);
294 os << std::forward<Arg>(arg);
295 auto len = std::strlen(replace.str);
296 return gformat(os, replace, next + len, std::forward<Args>(args)...);
297}

305std::string generic_output(unsigned int features,
306 const struct dlg_origin& origin, StringParam string,
307 const struct dlg_style styles[6]) {
308 std::size_t size;
309 dlg_generic_output_buf(nullptr, &size, features, &origin,
310 string.str, styles);
311 std::string ret(++size, ' ');
312
313 // NOTE: this might (theoretically) cause problems before C++17
314 // but this function should return a string and all compilers
315 // make it work. If you want to go sure, just use C++17
316 // where it is guaranteed that this works (everybody is
317 // using it anyways so shit will blow up if this fails)
318 dlg_generic_output_buf(&ret[0], &size, features, &origin,
319 string.str, styles);
320 ret.pop_back(); // terminating null-char
321 return ret;
322}
