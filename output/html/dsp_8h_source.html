174#define WEBP_DSP_INIT(func) do { \
175 static volatile VP8CPUInfo func ## _last_cpuinfo_used = \
176 (VP8CPUInfo)&func ## _last_cpuinfo_used; \
177 static pthread_mutex_t func ## _lock = PTHREAD_MUTEX_INITIALIZER; \
178 if (pthread_mutex_lock(&func ## _lock)) break; \
179 if (func ## _last_cpuinfo_used != VP8GetCPUInfo) func(); \
180 func ## _last_cpuinfo_used = VP8GetCPUInfo; \
181 (void)pthread_mutex_unlock(&func ## _lock); \
182} while (0)
183#else // !(defined(WEBP_USE_THREAD) && !defined(_WIN32))
184#define WEBP_DSP_INIT(func) do { \
185 static volatile VP8CPUInfo func ## _last_cpuinfo_used = \
186 (VP8CPUInfo)&func ## _last_cpuinfo_used; \
187 if (func ## _last_cpuinfo_used == VP8GetCPUInfo) break; \
188 func(); \
189 func ## _last_cpuinfo_used = VP8GetCPUInfo; \
190} while (0)
191#endif // defined(WEBP_USE_THREAD) && !defined(_WIN32)

193// Defines an Init + helper function that control multiple initialization of
194// function pointers / tables.
195/* Usage:
196 WEBP_DSP_INIT_FUNC(InitFunc) {
197 ...function body
198 }
199*/
200#define WEBP_DSP_INIT_FUNC(name) \
201 static WEBP_TSAN_IGNORE_FUNCTION void name ## _body(void); \
202 WEBP_TSAN_IGNORE_FUNCTION void name(void) { \
203 WEBP_DSP_INIT(name ## _body); \
204 } \
205 static WEBP_TSAN_IGNORE_FUNCTION void name ## _body(void)
