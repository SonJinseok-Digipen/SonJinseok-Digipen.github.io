65 void populate_buffer (hb_buffer_t *buffer, const char *text, int text_len,
66 const char *text_before, const char *text_after)
67 {
68 hb_buffer_clear_contents (buffer);
69 if (text_before) {
70 unsigned int len = strlen (text_before);
71 hb_buffer_add_utf8 (buffer, text_before, len, len, 0);
72 }
73 hb_buffer_add_utf8 (buffer, text, text_len, 0, text_len);
74 if (text_after) {
75 hb_buffer_add_utf8 (buffer, text_after, -1, 0, 0);
76 }
77
78 if (!utf8_clusters) {
79 /* Reset cluster values to refer to Unicode character index
80 * instead of UTF-8 index. */
81 unsigned int num_glyphs = hb_buffer_get_length (buffer);
82 hb_glyph_info_t *info = hb_buffer_get_glyph_infos (buffer, nullptr);
83 for (unsigned int i = 0; i < num_glyphs; i++)
84 {
85 info->cluster = i;
86 info++;
87 }
88 }
89
90 setup_buffer (buffer);
91 }

146static gboolean
147parse_shapers (const char *name G_GNUC_UNUSED,
148 const char *arg,
149 gpointer data,
150 GError **error)
151{
152 shape_options_t *shape_opts = (shape_options_t *) data;
153 char **shapers = g_strsplit (arg, ",", 0);
154
155 for (char **shaper = shapers; *shaper; shaper++)
156 {
157 bool found = false;
158 for (const char **hb_shaper = hb_shape_list_shapers (); *hb_shaper; hb_shaper++) {
159 if (strcmp (*shaper, *hb_shaper) == 0)
160 {
161 found = true;
162 break;
163 }
164 }
165 if (!found)
166 {
167 g_set_error (error, G_OPTION_ERROR, G_OPTION_ERROR_BAD_VALUE,
168 "Unknown or unsupported shaper: %s", *shaper);
169 g_strfreev (shapers);
170 return false;
171 }
172 }
173
174 g_strfreev (shape_opts->shapers);
175 shape_opts->shapers = shapers;
176 return true;
177}

192static gboolean
193parse_features (const char *name G_GNUC_UNUSED,
194 const char *arg,
195 gpointer data,
196 GError **error G_GNUC_UNUSED)
197{
198 shape_options_t *shape_opts = (shape_options_t *) data;
199 char *s = (char *) arg;
200 size_t l = strlen (s);
201 char *p;
202
203 shape_opts->num_features = 0;
204 g_free (shape_opts->features);
205 shape_opts->features = nullptr;
206
207 /* if the string is quoted, strip the quotes */
208 if (s[0] == s[l - 1] && (s[0] == '\"' || s[0] == '\''))
209 {
210 s[l - 1] = '\0';
211 s++;
212 }
213
214 if (!*s)
215 return true;
216
217 /* count the features first, so we can allocate memory */
218 p = s;
219 do {
220 shape_opts->num_features++;
221 p = strchr (p, ',');
222 if (p)
223 p++;
224 } while (p);
225
226 shape_opts->features = (hb_feature_t *) calloc (shape_opts->num_features, sizeof (*shape_opts->features));
227 if (!shape_opts->features)
228 return false;
229
230 /* now do the actual parsing */
231 p = s;
232 shape_opts->num_features = 0;
233 while (p && *p) {
234 char *end = strchr (p, ',');
235 if (hb_feature_from_string (p, end ? end - p : -1, &shape_opts->features[shape_opts->num_features]))
236 shape_opts->num_features++;
237 p = end ? end + 1 : nullptr;
238 }
239
240 return true;
241}
